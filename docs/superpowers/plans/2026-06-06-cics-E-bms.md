**Goal:** BMS map loader (via bms-tools git submodule), SEND MAP / RECEIVE MAP / SEND TEXT builtins and lowering in `CicsLoweringStrategy`, and an end-to-end integration test that drives the CardDemo sign-on → main menu conversational flow.

**Architecture:** bms-tools (Python library in `vendor/bms-tools/`) converts `.bms` copybook definitions to Python map objects. `BmsLoader` indexes these maps by name. At runtime, `__cics_send_map` enqueues a rendered screen dict to `screen_queue`; `__cics_receive_map` blocks on `input_queue.get()` and writes field values back to the mapped COBOL region. `__cics_send_text` enqueues plain text. The dispatcher's `run_cics()` exposes `screen_queue` and `input_queue` to the caller, which can provide a test driver or a terminal UI.

```python
"""BMS map loader — loads BMS mapset definitions for SEND MAP / RECEIVE MAP."""

from __future__ import annotations

import logging
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any

logger = logging.getLogger(__name__)


@dataclass
class BmsField:
    offset: int
    length: int
    attr: int = 0x00  # BMS attribute byte (default: unprotected, alphanumeric)


@dataclass
class BmsMap:
    name: str
    fields: dict[str, BmsField] = field(default_factory=dict)

    def to_dict(self) -> dict[str, Any]:
        return {
            "name": self.name,
            "fields": {
                fname: {"offset": f.offset, "length": f.length, "attr": f.attr}
                for fname, f in self.fields.items()
            },
        }

    def extract_fields(self, region: bytes) -> dict[str, bytes]:
        """Extract named field values from a map region (byte slice)."""
        result: dict[str, bytes] = {}
        for fname, fdef in self.fields.items():
            start = fdef.offset
            end = start + fdef.length
            result[fname] = region[start:end] if end <= len(region) else b""
        return result

    def write_fields(self, region: bytearray, values: dict[str, bytes]) -> None:
        """Write named field values into a map region (bytearray, mutated in-place)."""
        for fname, fval in values.items():
            fdef = self.fields.get(fname)
            if fdef is None:
                continue
            start = fdef.offset
            end = start + fdef.length
            padded = fval[:fdef.length].ljust(fdef.length, b" ")
            region[start:end] = padded[:fdef.length]


class BmsLoader:
    """Loads BMS maps from a directory of .bms files or stub registrations."""

    def __init__(self, maps_dir: Path | None) -> None:
        self._maps: dict[str, BmsMap] = {}
        if maps_dir is not None and maps_dir.exists():
            self._load_dir(maps_dir)

    def _load_dir(self, maps_dir: Path) -> None:
        """Load .bms files from directory. Best-effort — errors are logged and skipped."""
        for bms_file in maps_dir.glob("*.bms"):
            try:
                self._parse_bms_file(bms_file)
            except Exception as exc:
                logger.warning("BMS: failed to load %s: %s", bms_file, exc)

    def _parse_bms_file(self, path: Path) -> None:
        """Minimal .bms parser — extracts DFHMDF field definitions."""
        content = path.read_text(encoding="utf-8", errors="replace").upper()
        import re
        current_map: str | None = None
        fields: dict[str, BmsField] = {}
        # DFHMSD TYPE=MAP defines a mapset; DFHMDI defines a map
        for m in re.finditer(r"(\w+)\s+DFHMDI\b", content):
            current_map = m.group(1)
            fields = {}
        for m in re.finditer(
            r"(\w+)\s+DFHMDF\s+.*?POS=\((\d+),(\d+)\).*?LENGTH=(\d+)", content
        ):
            fname, row, col, length = m.group(1), int(m.group(2)), int(m.group(3)), int(m.group(4))
            # Flat offset = (row - 1) * 80 + (col - 1); 80-col 3270 screen approximation
            offset = (row - 1) * 80 + (col - 1)
            fields[fname] = BmsField(offset=offset, length=length)
        if current_map and fields:
            self._maps[current_map] = BmsMap(name=current_map, fields=fields)
            logger.info("BMS: loaded map %s (%d fields)", current_map, len(fields))

    def register_stub(self, map_name: str, bms_map: BmsMap) -> None:
        """Register a manually constructed stub map (for tests or pre-defined maps)."""
        self._maps[map_name.upper()] = bms_map

    def get(self, map_name: str) -> BmsMap | None:
        return self._maps.get(map_name.upper().strip("'\" "))
```

```python
"""BMS screen builtins: SEND MAP, RECEIVE MAP, SEND TEXT."""

from __future__ import annotations

import logging
import queue
from typing import TYPE_CHECKING

from interpreter.cics.bms.loader import BmsLoader
from interpreter.types.typed_value import TypedValue
from interpreter.vm.vm_types import BuiltinResult, VMState

logger = logging.getLogger(__name__)


def _map_name(args: list[TypedValue], idx: int = 0) -> str:
    return str(args[idx].value).strip("'\" ") if len(args) > idx else ""


def _region_bytes(args: list[TypedValue], idx: int = 2) -> bytes:
    v = args[idx].value if len(args) > idx else b""
    return bytes(v) if isinstance(v, (bytes, bytearray, list)) else str(v).encode()


def make_send_map_builtin(loader: BmsLoader, screen_queue: queue.Queue) -> object:
    def __cics_send_map(args: list[TypedValue], vm: VMState) -> BuiltinResult:
        map_name = _map_name(args, 0)
        region = _region_bytes(args, 2)
        bms_map = loader.get(map_name)
        if bms_map is None:
            logger.warning("SEND MAP: unknown map %s", map_name)
            screen_queue.put({"map": map_name, "fields": {}, "raw": region})
            return BuiltinResult(value=None)
        fields = bms_map.extract_fields(region)
        screen_queue.put({
            "map": map_name,
            "fields": {k: v.decode("cp037", errors="replace").rstrip() for k, v in fields.items()},
            "raw": region,
        })
        return BuiltinResult(value=None)

    return __cics_send_map


def make_receive_map_builtin(loader: BmsLoader, input_queue: queue.Queue) -> object:
    def __cics_receive_map(args: list[TypedValue], vm: VMState) -> BuiltinResult:
        map_name = _map_name(args, 0)
        region = bytearray(_region_bytes(args, 2))
        bms_map = loader.get(map_name)
        if bms_map is None:
            logger.warning("RECEIVE MAP: unknown map %s — blocking on input_queue", map_name)
        # Block until input arrives from the terminal or test driver
        try:
            field_values: dict[str, bytes | str] = input_queue.get(timeout=30)
        except queue.Empty:
            logger.warning("RECEIVE MAP: timeout waiting for input")
            return BuiltinResult(value=bytes(region))
        if bms_map is not None:
            encoded = {
                k: (v if isinstance(v, bytes) else v.encode("cp037", errors="replace"))
                for k, v in field_values.items()
            }
            bms_map.write_fields(region, encoded)
        return BuiltinResult(value=bytes(region))

    return __cics_receive_map


def make_send_text_builtin(screen_queue: queue.Queue) -> object:
    def __cics_send_text(args: list[TypedValue], vm: VMState) -> BuiltinResult:
        text = str(args[0].value) if args else ""
        screen_queue.put({"type": "text", "text": text})
        return BuiltinResult(value=None)

    return __cics_send_text
```
