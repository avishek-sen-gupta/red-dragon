> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Decouple TUI panels from domain types via MVVM: view dataclasses, builder functions, and framework-free ViewModels.

**Architecture:** Per-panel view dataclasses in `viz/views/` with pure builder functions. Two ViewModels (`ExecutionViewModel`, `OverviewViewModel`) in `viz/viewmodels/` own state and expose views. Screens become thin sync glue. Panels import only from `viz/views/`.

**Tech Stack:** Python 3.13+, Textual TUI framework, frozen dataclasses

```python
    register_writes: dict | None = None,
    var_writes: dict | None = None,
```

```python
# viz/views/vm_state_view.py
"""VMState view — display-ready VM state for the VMStatePanel."""

from __future__ import annotations

from dataclasses import dataclass

from interpreter.trace_types import TraceStep
from interpreter.types.typed_value import TypedValue
from interpreter.vm.vm_types import HeapObject, Pointer, SymbolicValue


@dataclass(frozen=True)
class FrameView:
    """Display-ready stack frame."""

    function_name: str
    is_current: bool
    locals: list[tuple[str, str]]
    registers: list[tuple[str, str]]
    changed_vars: frozenset[str]
    changed_regs: frozenset[str]


@dataclass(frozen=True)
class HeapObjectView:
    """Display-ready heap object."""

    address: str
    type_hint: str
    fields: list[tuple[str, str]]
    changed_fields: frozenset[str]


@dataclass(frozen=True)
class VMStateView:
    """Display-ready VM state snapshot."""

    frames: list[FrameView]
    heap: list[HeapObjectView]


def format_value(val: object) -> str:
    """Format a VM value for display."""
    if isinstance(val, TypedValue):
        return format_value(val.value)
    if isinstance(val, SymbolicValue):
        return f"sym:{val.name}"
    if isinstance(val, Pointer):
        return f"→ {val.base}[{val.offset}]"
    if isinstance(val, HeapObject):
        return f"<{val.type_hint or 'obj'}>"
    if isinstance(val, str) and len(val) > 40:
        return f'"{val[:37]}..."'
    if isinstance(val, str):
        return f'"{val}"'
    return repr(val)


def build_vm_state_view(step: TraceStep) -> VMStateView:
    """Convert a TraceStep's VM state into a display-ready VMStateView."""
    vm = step.vm_state
    update = step.update
    changed_regs = frozenset(str(r) for r in update.register_writes)
    changed_vars = frozenset(str(v) for v in update.var_writes)
    changed_heap = {
        (str(hw.obj_addr), str(hw.field)) for hw in update.heap_writes
    }

    frames: list[FrameView] = []
    for i, frame in enumerate(reversed(vm.call_stack)):
        is_current = i == 0
        frames.append(
            FrameView(
                function_name=str(frame.function_name),
                is_current=is_current,
                locals=[
                    (str(v), format_value(val))
                    for v, val in sorted(
                        frame.local_vars.items(), key=lambda kv: str(kv[0])
                    )
                ],
                registers=[
                    (str(r), format_value(val))
                    for r, val in sorted(
                        frame.registers.items(), key=lambda kv: str(kv[0])
                    )
                ]
                if is_current
                else [],
                changed_vars=changed_vars,
                changed_regs=changed_regs,
            )
        )

    heap: list[HeapObjectView] = []
    for addr, obj in sorted(vm.heap_items(), key=lambda kv: str(kv[0])):
        addr_str = str(addr)
        heap.append(
            HeapObjectView(
                address=addr_str,
                type_hint=obj.type_hint or "",
                fields=[
                    (str(f), format_value(v))
                    for f, v in sorted(
                        obj.fields.items(), key=lambda kv: str(kv[0])
                    )
                ],
                changed_fields=frozenset(
                    f for a, f in changed_heap if a == addr_str
                ),
            )
        )

    return VMStateView(frames=frames, heap=heap)
```

```python
# viz/views/step_view.py
"""Step view — display-ready step summary for the StepPanel."""

from __future__ import annotations

from dataclasses import dataclass

from interpreter.trace_types import TraceStep
from interpreter.vm.vm_types import VOID_RETURN


@dataclass(frozen=True)
class DeltaEntry:
    """A single state change in a step."""

    kind: str
    label: str
    value: str


@dataclass(frozen=True)
class StepView:
    """Display-ready step summary."""

    step_number: int
    total_steps: int
    instruction_text: str
    deltas: list[DeltaEntry]
    next_label: str | None
    call_push: str | None
    call_pop_value: str | None
    reasoning: str
    used_llm: bool
    playing: bool


def _short_val(val: object) -> str:
    """Short representation of a value for the delta summary."""
    s = repr(val)
    return s[:60] + "..." if len(s) > 60 else s


def build_step_view(
    step: TraceStep, total_steps: int, playing: bool
) -> StepView:
    """Convert a TraceStep into a display-ready StepView."""
    update = step.update
    deltas: list[DeltaEntry] = []

    for reg, val in update.register_writes.items():
        deltas.append(DeltaEntry(kind="reg", label=str(reg), value=_short_val(val)))

    for var, val in update.var_writes.items():
        deltas.append(DeltaEntry(kind="var", label=str(var), value=_short_val(val)))

    for hw in update.heap_writes:
        deltas.append(
            DeltaEntry(
                kind="heap",
                label=f"{hw.obj_addr}.{hw.field}",
                value=_short_val(hw.value),
            )
        )

    call_push_name = str(update.call_push.function_name) if update.call_push else None

    call_pop_value = None
    if update.call_pop:
        ret = update.return_value
        call_pop_value = _short_val(ret) if ret is not None and ret != VOID_RETURN else "(void)"

    return StepView(
        step_number=step.step_index + 1,
        total_steps=total_steps,
        instruction_text=str(step.instruction),
        deltas=deltas,
        next_label=str(update.next_label) if update.next_label else None,
        call_push=call_push_name,
        call_pop_value=call_pop_value,
        reasoning=update.reasoning,
        used_llm=step.used_llm,
        playing=playing,
    )
```

```python
        current_idx = step.instruction_index

        label_str = str(label)

                current_instruction_index=current_idx if is_current else -1,
                is_label_opcode=[
                    inst.opcode == Opcode.LABEL for inst in block.instructions
                ],

        )
```

```python
"""CFG view — display-ready block graph for the CFGPanel."""

from __future__ import annotations

from dataclasses import dataclass

from interpreter.cfg_types import CFG
from interpreter.ir import Opcode
from interpreter.trace_types import TraceStep


@dataclass(frozen=True)
class CFGBlockView:
    """Display-ready CFG block."""

    label: str
    is_current: bool
    instruction_count: int
    terminator: str
    successors: list[str]
    edge_labels: list[str]


@dataclass(frozen=True)
class CFGView:
    """Display-ready CFG."""

    blocks: list[CFGBlockView]


def _block_terminator(block) -> str:
    """Get a short description of the block's terminating instruction."""
    if not block.instructions:
        return ""
    last = block.instructions[-1]
    if last.opcode == Opcode.BRANCH_IF:
        return "branch_if"
    if last.opcode == Opcode.BRANCH:
        return "branch"
    if last.opcode == Opcode.RETURN:
        return "return"
    if last.opcode == Opcode.THROW:
        return "throw"
    return ""


def _edge_label(block, succ_idx: int) -> str:
    """Label for a CFG edge (T/F for conditional branches)."""
    if not block.instructions:
        return ""
    last = block.instructions[-1]
    if last.opcode == Opcode.BRANCH_IF and len(block.successors) == 2:
        return "T" if succ_idx == 0 else "F"
    return ""


def build_cfg_view(cfg: CFG, step: TraceStep | None = None) -> CFGView:
    """Convert a CFG (+ optional step) into a display-ready CFGView."""
    current_block = str(step.block_label) if step else ""

    blocks: list[CFGBlockView] = []
    for label, block in cfg.blocks.items():
        label_str = str(label)
        blocks.append(
            CFGBlockView(
                label=label_str,
                is_current=label_str == current_block,
                instruction_count=len(block.instructions),
                terminator=_block_terminator(block),
                successors=[str(s) for s in block.successors],
                edge_labels=[_edge_label(block, i) for i in range(len(block.successors))],
            )
        )
    return CFGView(blocks=blocks)
```

```python
# viz/views/dataflow_view.py
"""Dataflow views — display-ready dataflow summaries and call chains."""

from __future__ import annotations

from dataclasses import dataclass

from interpreter.cfg_types import CFG
from interpreter.interprocedural.types import (
    CallGraph,
    FieldEndpoint,
    FlowEndpoint,
    FunctionEntry,
    FunctionSummary,
    InterproceduralResult,
    ReturnEndpoint,
    SummaryKey,
    VariableEndpoint,
)
from interpreter.interprocedural.call_graph import _build_block_to_function
from interpreter.instructions import (
    CallFunction,
    CallMethod,
    DeclVar,
    LoadVar,
    StoreVar,
)
from interpreter.ir import CodeLabel


# ── Summary view ─────────────────────────────────────────────────────


@dataclass(frozen=True)
class FunctionSummaryView:
    """Display-ready function summary."""

    label: str
    params: str
    callers: str
    callees: str
    flows: list[str]


@dataclass(frozen=True)
class DataflowSummaryView:
    """Display-ready call graph + per-function summaries."""

    functions: list[FunctionSummaryView]


# ── Graph view ───────────────────────────────────────────────────────


@dataclass(frozen=True)
class ChainNodeView:
    """Display-ready call chain tree node."""

    label: str
    children: list[ChainNodeView]


@dataclass(frozen=True)
class DataflowGraphView:
    """Display-ready interprocedural call chains."""

    top_level_chains: list[ChainNodeView]
    fallback_chains: list[ChainNodeView]


# ── Helpers (moved from panels) ──────────────────────────────────────


def _render_endpoint(ep: FlowEndpoint) -> str:
    if isinstance(ep, VariableEndpoint):
        return ep.name
    if isinstance(ep, ReturnEndpoint):
        return f"Return({ep.function.label})"
    if isinstance(ep, FieldEndpoint):
        return f"Field({ep.base.name}.{ep.field})"
    return str(ep)


def _build_callers(func: FunctionEntry, call_graph: CallGraph) -> set[str]:
    return {str(site.caller.label) for site in call_graph.call_sites if func in site.callees}


def _build_callees(func: FunctionEntry, call_graph: CallGraph) -> set[str]:
    return {
        str(callee.label)
        for site in call_graph.call_sites
        if site.caller == func
        for callee in site.callees
    }


def _merge_flows(
    func: FunctionEntry, summaries: dict[SummaryKey, FunctionSummary]
) -> set[tuple[FlowEndpoint, FlowEndpoint]]:
    merged: set[tuple[FlowEndpoint, FlowEndpoint]] = set()
    for key, summary in summaries.items():
        if key.function == func:
            merged.update(summary.flows)
    return merged


def _trace_reg_to_var(reg: str, cfg: CFG, block_label) -> str:
    block = cfg.blocks[block_label]
    load_match = next(
        (
            str(t.name)
            for inst in block.instructions
            if isinstance((t := inst), LoadVar) and str(t.result_reg) == reg
        ),
        "",
    )
    if load_match:
        return load_match
    store_match = next(
        (
            str(t.name)
            for inst in block.instructions
            if isinstance((t := inst), (DeclVar, StoreVar)) and str(t.value_reg) == reg
        ),
        "",
    )
    return store_match if store_match else reg


# ── Summary builder ──────────────────────────────────────────────────


def build_dataflow_summary_view(result: InterproceduralResult) -> DataflowSummaryView:
    """Build display-ready summaries for all functions."""
    call_graph = result.call_graph
    sorted_functions = sorted(call_graph.functions, key=lambda f: str(f.label))

    functions: list[FunctionSummaryView] = []
    for func in sorted_functions:
        params_str = ", ".join(func.params) if func.params else "(none)"
        callers = _build_callers(func, call_graph)
        callers_str = ", ".join(sorted(callers)) if callers else "(none)"
        callees = _build_callees(func, call_graph)
        callees_str = ", ".join(sorted(callees)) if callees else "(none)"
        flows = _merge_flows(func, result.summaries)
        flow_strs = [
            f"{_render_endpoint(src)} \u2192 {_render_endpoint(dst)}"
            for src, dst in sorted(flows, key=lambda f: _render_endpoint(f[0]))
        ]
        functions.append(
            FunctionSummaryView(
                label=str(func.label),
                params=params_str,
                callers=callers_str,
                callees=callees_str,
                flows=flow_strs,
            )
        )
    return DataflowSummaryView(functions=functions)


# ── Graph builder ────────────────────────────────────────────────────


def _build_chain_view(
    func_entry: FunctionEntry,
    call_graph: CallGraph,
    summaries: dict[SummaryKey, FunctionSummary],
    cfg: CFG,
    visited: set[str],
) -> list[ChainNodeView]:
    if str(func_entry.label) in visited:
        return [ChainNodeView(label="[recursive — see above]", children=[])]
    visited = visited | {str(func_entry.label)}

    flows = _merge_flows(func_entry, summaries)
    inner_sites = [s for s in call_graph.call_sites if s.caller == func_entry]

    def _nodes_for_param(param: str) -> list[ChainNodeView]:
        inner_calls = [
            (site, arg_op)
            for site in inner_sites
            for arg_op in site.arg_operands
            if _trace_reg_to_var(arg_op, cfg, site.location.block_label) == param
        ]
        call_nodes = []
        for site, arg_op in inner_calls:
            for callee in site.callees:
                param_map = {
                    formal: _trace_reg_to_var(actual_reg, cfg, site.location.block_label)
                    for formal, actual_reg in zip(callee.params, site.arg_operands)
                }
                label = f"{param} → {callee.label}({', '.join(f'{p}={v}' for p, v in param_map.items())})"
                children = _build_chain_view(callee, call_graph, summaries, cfg, visited)
                call_nodes.append(ChainNodeView(label=label, children=children))
        return_nodes = [
            ChainNodeView(label=f"{param} → return({func_entry.label})", children=[])
            for src, dst in flows
            if isinstance(src, VariableEndpoint)
            and src.name == param
            and isinstance(dst, ReturnEndpoint)
        ]
        field_nodes = [
            ChainNodeView(
                label=f"{param} → Field({dst.base.name}.{dst.field})", children=[]
            )
            for src, dst in flows
            if isinstance(src, VariableEndpoint)
            and src.name == param
            and isinstance(dst, FieldEndpoint)
        ]
        return call_nodes + return_nodes + field_nodes

    return [node for param in func_entry.params for node in _nodes_for_param(param)]


def build_dataflow_graph_view(
    result: InterproceduralResult, cfg: CFG
) -> DataflowGraphView:
    """Build display-ready call chain trees."""
    call_graph = result.call_graph
    func_by_label = {f.label: f for f in call_graph.functions}
    func_by_name: dict[str, CodeLabel] = {str(f.label): f.label for f in call_graph.functions}
    func_by_name.update(
        {
            (
                f.label.extract_name("func_")
                if f.label.starts_with("func_") and "_" in str(f.label)[5:]
                else str(f.label)
            ): f.label
            for f in call_graph.functions
        }
    )

    non_func_blocks = {
        label for label in cfg.blocks if label.starts_with("end_") or label == "entry"
    }

    top_calls = []
    for label in non_func_blocks:
        for idx, inst in enumerate(cfg.blocks[label].instructions):
            t = inst
            if isinstance(t, CallFunction):
                callee_name = str(t.func_name)
                arg_ops = tuple(str(a) for a in t.args)
            elif isinstance(t, CallMethod):
                callee_name = str(t.method_name)
                arg_ops = tuple(str(a) for a in t.args)
            else:
                continue
            result_var = (
                _trace_reg_to_var(str(t.result_reg), cfg, label)
                if t.result_reg.is_present()
                else ""
            )
            callee_label = func_by_name.get(callee_name, callee_name)
            callee_entry = func_by_label.get(callee_label)
            if callee_entry is None:
                callee_entry = next(
                    (
                        f
                        for f in call_graph.functions
                        if f.label.extract_name("func_") == callee_name
                        if f.label.starts_with("func_") and "_" in str(f.label)[5:]
                    ),
                    None,
                )
            root_label = f"{callee_name}({', '.join(arg_ops)}) → {result_var}"
            if callee_entry is None:
                top_calls.append(ChainNodeView(label=root_label + " [unresolved]", children=[]))
            else:
                children = _build_chain_view(
                    callee_entry, call_graph, result.summaries, cfg, set()
                )
                top_calls.append(ChainNodeView(label=root_label, children=children))

    fallback: list[ChainNodeView] = []
    if not top_calls:
        for func in sorted(call_graph.functions, key=lambda f: str(f.label)):
            children = _build_chain_view(
                func, call_graph, result.summaries, cfg, set()
            )
            fallback.append(
                ChainNodeView(
                    label=f"{func.label}({', '.join(func.params)})",
                    children=children,
                )
            )

    return DataflowGraphView(top_level_chains=top_calls, fallback_chains=fallback)
```

```python
from __future__ import annotations

from dataclasses import dataclass
from pathlib import Path

from viz.panels.import_graph_panel import render_import_graph

def _group_entry_points(
    topo_order: list[Path],
    func_symbol_table: dict[CodeLabel, FuncRef],
) -> list[tuple[Path, list[FuncRef]]]:
    """Group function refs by module, ordered by topo_order."""
    result: list[tuple[Path, list[FuncRef]]] = []
    for path in topo_order:
        stem = path.stem
        matched: list[FuncRef] = []
        for label, ref in func_symbol_table.items():
            label_str = str(label)
            if label_str.startswith(stem + ".") or label_str.startswith(stem + "_"):
                matched.append(ref)
        result.append((path, matched))
    return result


def build_overview_view(
    result: ProjectPipelineResult,
    project_root: Path,
) -> OverviewView:
    """Build display-ready overview from pipeline result."""
    exports: dict[Path, tuple[int, int]] = {}
    for path, module in result.linked.modules.items():
        fn_count = len(module.exports.functions)
        var_count = len(module.exports.variables)
        exports[path] = (fn_count, var_count)

    import_graph_text = render_import_graph(
        result.topo_order,
        result.linked.import_graph,
        exports,
        project_root,
    )

    # Build entry point options
    options: list[EntryPointOption] = []
    actual_idx = 0

    # First option: top-level execution
    options.append(
        EntryPointOption(
            display_text="[t] Top-level execution",
            is_separator=False,
            is_disabled=False,
            option_index=actual_idx,
        )
    )
    actual_idx += 1

    grouped = _group_entry_points(result.topo_order, result.linked.func_symbol_table)
    for path, funcs in grouped:
        rel = (
            path.relative_to(project_root)
            if path.is_relative_to(project_root)
            else path
        )
        # Separator
        options.append(
            EntryPointOption(display_text="", is_separator=True, is_disabled=False, option_index=actual_idx)
        )
        actual_idx += 1
        # Module header
        options.append(
            EntryPointOption(display_text=f"  {rel}", is_separator=False, is_disabled=True, option_index=actual_idx)
        )
        actual_idx += 1
        if not funcs:
            options.append(
                EntryPointOption(
                    display_text="    (no functions)",
                    is_separator=False,
                    is_disabled=True,
                    option_index=actual_idx,
                )
            )
            actual_idx += 1
        else:
            for ref in funcs:
                options.append(
                    EntryPointOption(
                        display_text=f"    {ref.name}()",
                        is_separator=False,
                        is_disabled=False,
                        option_index=actual_idx,
                    )
                )
                actual_idx += 1

    return OverviewView(
        import_graph_text=import_graph_text,
        module_count=len(result.topo_order),
        entry_points=options,
    )
```

```python
"""ExecutionViewModel — framework-free state + commands + views for execution screen."""

from __future__ import annotations

from collections.abc import Callable
from pathlib import Path

from viz.pipeline import ASTNode
from viz.project_pipeline import ProjectPipelineResult, lookup_module_for_index
from viz.views.cfg_view import CFGView, build_cfg_view
from viz.views.dataflow_view import (
    DataflowGraphView,
    DataflowSummaryView,

    build_dataflow_summary_view,
)

from viz.views.vm_state_view import VMStateView, build_vm_state_view


class ExecutionViewModel:
    """Owns execution state. Exposes views. Handles navigation commands.

    Framework-free: no Textual imports. Fully unit-testable.
    """

    def __init__(self, result: ProjectPipelineResult, project_root: Path) -> None:
        self._result = result
        self._project_root = project_root
        self._step_index = 0
        self._playing = False
        self._dataflow_mode = False
        self._highlighted_block: str | None = None
        self._current_module: Path | None = (
            result.topo_order[0] if result.topo_order else None
        )
        self._previous_module: Path | None = self._current_module
        self._on_change: Callable[[], None] | None = None

        # Cache dataflow views (built once, not per-step)
        self._dataflow_summary_cache: DataflowSummaryView | None = None
        self._dataflow_graph_cache: DataflowGraphView | None = None

    # ── Binding ──────────────────────────────────────────────────────

    def bind(self, callback: Callable[[], None]) -> None:
        self._on_change = callback

    def _notify(self) -> None:
        if self._on_change:
            self._on_change()

    # ── Commands ─────────────────────────────────────────────────────

    def step_forward(self) -> None:
        if self._step_index < self.total_steps - 1:
            self._step_index += 1
            self._check_module_switch()
            self._notify()

    def step_backward(self) -> None:
        if self._step_index > 0:
            self._step_index -= 1
            self._check_module_switch()
            self._notify()

    def step_first(self) -> None:
        self._step_index = 0
        self._check_module_switch()
        self._notify()

    def step_last(self) -> None:
        self._step_index = max(0, self.total_steps - 1)
        self._check_module_switch()
        self._notify()

    def toggle_play(self) -> None:
        self._playing = not self._playing
        self._notify()

    def toggle_dataflow(self) -> None:
        self._dataflow_mode = not self._dataflow_mode
        self._notify()

    def highlight_block(self, label: str | None) -> None:
        self._highlighted_block = label
        self._notify()

    # ── Computed state ───────────────────────────────────────────────

    @property
    def step_index(self) -> int:
        return self._step_index

    @property
    def total_steps(self) -> int:
        return len(self._steps)

    @property
    def playing(self) -> bool:
        return self._playing

    @property
    def dataflow_mode(self) -> bool:
        return self._dataflow_mode

    @property
    def module_changed(self) -> bool:
        return self._current_module != self._previous_module

    @property
    def current_module_source(self) -> str:
        if self._current_module is None:
            return ""
        return self._result.module_sources.get(self._current_module, "")

    @property
    def current_module_ast(self) -> ASTNode | None:
        if self._current_module is None:
            return None
        return self._result.module_asts.get(self._current_module)

    @property
    def current_module_name(self) -> str:
        if self._current_module is None:
            return "?"
        try:
            return str(self._current_module.relative_to(self._project_root))
        except ValueError:
            return self._current_module.name

    # ── View properties ──────────────────────────────────────────────

    @property
    def vm_state(self) -> VMStateView:
        return build_vm_state_view(self._current_step)

    @property
    def cfg(self) -> CFGView:
        return build_cfg_view(self._cfg, self._current_step)

    @property
    def ir(self) -> IRView:
        return build_ir_view(self._cfg, self._current_step, self._highlighted_block)

    @property
    def source(self) -> SourceView:
        return build_source_view(self._current_step.instruction)

    @property
    def step(self) -> StepView:
        return build_step_view(self._current_step, self.total_steps, self._playing)

    @property
    def dataflow_summary(self) -> DataflowSummaryView | None:
        if self._result.interprocedural is None:
            return None
        if self._dataflow_summary_cache is None:
            self._dataflow_summary_cache = build_dataflow_summary_view(
                self._result.interprocedural
            )
        return self._dataflow_summary_cache

    @property
    def dataflow_graph(self) -> DataflowGraphView | None:
        if self._result.interprocedural is None:
            return None
        if self._dataflow_graph_cache is None:
            self._dataflow_graph_cache = build_dataflow_graph_view(
                self._result.interprocedural, self._cfg
            )
        return self._dataflow_graph_cache

    # ── Internal ─────────────────────────────────────────────────────

    @property
    def _steps(self):
        return self._result.trace.steps if self._result.trace else []

    @property
    def _current_step(self):
        return self._steps[self._step_index]

    @property
    def _cfg(self):
        return self._result.linked.merged_cfg

    def _check_module_switch(self) -> None:
        self._previous_module = self._current_module
        step = self._current_step
        inst_id = id(step.instruction)
        idx = self._result.instruction_to_index.get(inst_id)
        if idx is not None:
            module_path = lookup_module_for_index(self._result.module_ir_ranges, idx)
            if module_path:
                self._current_module = module_path
```

```python
"""OverviewViewModel — framework-free state + commands for project overview screen."""

from __future__ import annotations

from pathlib import Path

from interpreter.project.entry_point import EntryPoint
from interpreter.refs.func_ref import FuncRef
from viz.project_pipeline import ProjectPipelineResult
from viz.views.overview_view import OverviewView, build_overview_view


class OverviewViewModel:
    """Owns project overview state. Handles entry point selection.

    Framework-free: no Textual imports. Fully unit-testable.
    """

    def __init__(self, result: ProjectPipelineResult, project_root: Path) -> None:
        self._result = result
        self._project_root = project_root
        self._entry_point_map: dict[int, FuncRef | None] = {}
        self._overview_cache: OverviewView | None = None

    @property
    def overview(self) -> OverviewView:
        """Build display data for overview screen. Populates _entry_point_map."""
        if self._overview_cache is not None:
            return self._overview_cache

        view = build_overview_view(self._result, self._project_root)

        # Build index→FuncRef mapping from the entry point options
        # Option index 0 = top-level (None)
        self._entry_point_map[0] = None

        # Map selectable function options to their FuncRefs
        func_symbol_table = self._result.linked.func_symbol_table
        func_refs_by_name: dict[str, FuncRef] = {}
        for _label, ref in func_symbol_table.items():
            func_refs_by_name[str(ref.name)] = ref

        for ep in view.entry_points:
            if ep.is_separator or ep.is_disabled:
                continue
            if ep.option_index == 0:
                continue  # already mapped
            # Extract function name from display text "    helper()"
            text = ep.display_text.strip()
            if text.endswith("()"):
                func_name = text[:-2]
                ref = func_refs_by_name.get(func_name)
                if ref is not None:
                    self._entry_point_map[ep.option_index] = ref

        self._overview_cache = view
        return view

    def select_entry_point(self, option_index: int) -> EntryPoint | None:
        """Translate option index to domain EntryPoint. Returns None for top-level."""
        if option_index not in self._entry_point_map:
            return None
        func_ref = self._entry_point_map[option_index]
        if func_ref is None:
            return None
        name = func_ref.name
        return EntryPoint.function(lambda f, n=name: f.name == n)
```

```python
"""AST panel — collapsible tree view of the tree-sitter AST with current node highlighted."""

from __future__ import annotations

from textual.reactive import reactive
from textual.widgets import Tree

from viz.pipeline import ASTNode
from viz.views.source_view import SourceView


class ASTPanel(Tree):
    """Displays the tree-sitter AST as a collapsible tree widget."""

    current_view: reactive[SourceView | None] = reactive(None)

    def __init__(self, ast: ASTNode | None = None, **kwargs) -> None:
        super().__init__("AST", **kwargs)
        self._ast = ast

    def set_ast(self, ast: ASTNode) -> None:
        self._ast = ast
        self.root.remove_children()
        if self._ast:
            self._populate_tree(self.root, self._ast)
            self.root.expand()
            for child in self.root.children:
                child.expand()

    def on_mount(self) -> None:
        if self._ast:
            self._populate_tree(self.root, self._ast)
            self.root.expand()
            for child in self.root.children:
                child.expand()

    def _populate_tree(self, parent, ast_node: ASTNode) -> None:
        label = self._make_label(ast_node)
        if ast_node.children:
            tree_node = parent.add(label, data=ast_node)
            for child in ast_node.children:
                self._populate_tree(tree_node, child)
        else:
            parent.add_leaf(label, data=ast_node)

    def _make_label(self, ast_node: ASTNode) -> str:
        type_part = ast_node.node_type
        if not ast_node.children:
            text = ast_node.text.replace("\n", "\\n")
            if len(text) > 40:
                text = text[:37] + "..."
            return f"{type_part}: {text}"
        return type_part

    def watch_current_view(self, view: SourceView | None) -> None:
        if not view or view.highlight_start < 0:
            return
        target_line = view.highlight_start + 1  # 0-indexed → 1-indexed
        target_col = 0
        best_node = self._find_best_match(self.root, target_line, target_col)
        if best_node:
            best_node.expand()
            self.select_node(best_node)
            self.scroll_to_node(best_node)

    def _find_best_match(self, tree_node, target_line: int, target_col: int):
        best = None
        best_size = float("inf")
        for child in tree_node.children:
            ast_node = child.data
            if ast_node is None:
                continue
            if self._contains(ast_node, target_line, target_col):
                span_size = (
                    (ast_node.end_line - ast_node.start_line) * 1000
                    + ast_node.end_col
                    - ast_node.start_col
                )
                if span_size < best_size:
                    best = child
                    best_size = span_size
                deeper = self._find_best_match(child, target_line, target_col)
                if deeper and deeper.data:
                    deeper_ast = deeper.data
                    deeper_size = (
                        (deeper_ast.end_line - deeper_ast.start_line) * 1000
                        + deeper_ast.end_col
                        - deeper_ast.start_col
                    )
                    if deeper_size < best_size:
                        best = deeper
                        best_size = deeper_size
        return best

    def _contains(self, ast_node: ASTNode, line: int, col: int) -> bool:
        if line < ast_node.start_line or line > ast_node.end_line:
            return False
        if line == ast_node.start_line and col < ast_node.start_col:
            return False
        if line == ast_node.end_line and col > ast_node.end_col:
            return False
        return True
```
